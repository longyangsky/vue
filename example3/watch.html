    <div id="app"> <p> 苹果{{price}}块一斤</p> <p>{{message}}</p> <button @click="priceIncrease">加价</button> <button @click="priceReduction">降价</button> </div> 
            var vue = new Vue({
            el: '#app',
            data: {
                price: 5,
                message: '价格适中',
            },
            methods:{
                priceIncrease :function(){
                    if(this.price>=10){
                        return;
                    }
                    this.price++;
                },
                priceReduction:function(){
                    if(this.price<=3){
                        return;
                    }
                    this.price--;
                }
            },
            watch: {
                "price": function (newval, oldval) {
                    if (newval >= 6) {
                        this.message = "价格贵了"
                    } else if (newval>4&&newval<6) {
                        this.message = "价格适中";
                    } else {
                        this.message = "便宜";
                    }
                }
            }
        })
