extends扩展与mixins混合基本一样,不过extends接收对象而mixins接收的是数组

可以只用new Vue不用赋值得到对象

num:${num}

<div id="app"> <p>num:${num}</p> <button @click="add">add</button> </div> 
            var extendsObj = {
                updated: function () {
                    console.log('extends扩展更新');
                },
                methods:{
                    reduce:function(){
                        this.num--;
                    }
                }
            }
            var vue = new Vue({
                el: '#app',
                data: {
                    num: 1
                },
                updated: function () {
                    console.log('构造器更新');
                },
                methods: {
                    add: function () {
                        this.num++;
                    }
                },
                extends: extendsObj,
                delimiters: ["${", "}"]
            })
