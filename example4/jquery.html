与第三方框架使用

必须在vue构造器的mounted或updated方法中,不然无法获取正确的dom

{{num}}

<div id="app"> <p> {{num}}</p> <button @click="add">add</button> </div> 
            var  vue= new Vue({
                el:'#app',
                data:{
                   num:1
                },
                methods:{
                    add:function(){
                        this.num++;
                    }
                },
                mounted:function(){
                    $("#app").html("我是JQERY22")
                }
            })
