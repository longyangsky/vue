vue内置的$methods方法

$mount挂载,$destroy销毁,$forceUpdate强制刷新,$netxtTick接受上个事件回调

<div id="app"> </div> <p> <button onclick="udpate()">更新</button> <button onclick="reload()">刷新</button> <button onclick="destroy()">卸载</button> </p> 
        function reload(){
            vm.$forceUpdate(); 
        }
        function destroy(){
            vm.$destroy();
        }
        function udpate(){
            vm.message="三只小猪";
            //当修改了这个值就会触发nextTick
            vm.$nextTick(function(){ 
                console.log("update更新完后回调");
            })
        }
        var book=Vue.extend({
            template:'
<<{{message}}>>
',
            data:function(){
                return {
                    message:'小蝌蚪找妈妈'
                }
            },
            mounted:function(){
                console.log('已经被挂载');
            },
            destroyed:function(){
                console.log('已经被销毁');
            }
        })
        var vm = new book().$mount("#app")
