vue的componet组件

注意vue的组件一定是在vue作用域下也就是需要不过绑定vue后使用，组件必须要有根元素
  1. 定义全局组件Vue.component("标签名",{模板名:"模板字符串"}) 
                    Vue.component("ly",{
                    template:`
    全局组件你好
    `
                })
                 var  vue= new Vue({
                     el:'#app'
                 })
                 var  vue= new Vue({
                     el:'#app2'
                 })
  2. 内部定义为局部可以定义多个组件new Vue({ el:'绑定的标签', components:{ "模板名":{ template:"模板字符串" } } }) 
                        var vue3=new Vue({
                        el:'#app3',
                        components:{
                            "ly1":{
                                template:"
    我是局部ly
    "
                            },
                            "ly2":{
                                template:"
    我是局部ly2
    "
                            }
                        }
                    })
