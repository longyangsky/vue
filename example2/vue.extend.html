vue构造器扩展实例

有点类似模板new Extend后使用mount绑定到对应元素上
注意模板只有一个根节点 
            var book = Vue.extend({
                template:"hello world{{count}} ",
                data: function(){ //注意是function
                    return { 
                      count:222
                    }
                }
                
            })
            new book().$mount("#app")
            new book().$mount("book")
