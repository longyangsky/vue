vue的componet组件定义属性

  1. 直接属性赋值name="<<三只小猪>>"
    <div id="app"> <book name="<<三只小猪>>"></book> </div> 
     
                var  vue= new Vue({
                    el:'#app',  
                    components:{
                        "book":{
                            template:"
    这是一本书{{name}}
    ",
                            props:["name"]
                        }
                    }
                })
  2. 通过属性绑定 :name="name"
    <div id="app2"> <book :name="name"></book> </div> 
     
                var vue = new Vue({
                    el: '#app2',
                    data: {
                        name: '<<小蝌蚪找妈妈>>'
                    },
                    components: {
                        "book": {
                            template: "
    这是一本书{{name}}
    ",
                            props: ["name"]
                        }
                    }
                })
