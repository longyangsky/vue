vue的父子组件

构造器包含父组件，父组件包含子组件，顺序是这样，注意定义组件的顺序要对 
        var bookpublishing={
            template:'
出版社:教育出版社
'
         
        }
         var bookName={
                     template:`
这是一本书《{{name}}》
                        
                     
`, 
                     data:{
                         pub:"教育出版社"
                     },
                     props:["name"],
                     components:{
                         "bookpublishing":bookpublishing
                     }
                 }
         var  vue= new Vue({
             el:'#app',
             components:{ 
                 "book":bookName
             }
         })
