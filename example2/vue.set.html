vue.set构造器外赋值

由于javascript无法监视数组下标变化，必须使用vue.set赋值才能刷新dom

count为:{{count}}

arrCount为:{{arrCount}}

        function addCount() {
            vue.count++;
        } 
        function addArrCount() { 
            //直接修改是无效
            vue.arr[1]=222; 
        }
        function setArrCount(){
            Vue.set(vue.arrCount,1,333); 
            //修改数组
            Vue.set(outData.arrCount,2,22);
            //修改值
            Vue.set(outData,"count",4);
            Vue.set(vue,"count",555);
        }
        var outData = {
            count: 1,
            arrCount: [1, 5, 8, 9]
        }
        var vue = new Vue({
            el: '#app',
            data: outData
        })
