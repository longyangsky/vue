            var  vue= new Vue({
                el:'#app',
                data:{
                   score:1,
                   score2:2
                },
                methods:{
                    addScore:function(){
                        this.score++;
                    },
                    subScore:function(){
                        this.score--;
                    },
                    enterAddScore:function(){
                       this.score+=this.score2;
                    }
                }
            })
