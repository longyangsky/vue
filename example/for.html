v-for命令

放在需要循环的标签中 利用computed对data数据重新计算生成新的data
    循环输出数字数组
  • {{item}}
    • 循环输出josn数组
  • {{index+1}}.{{item.name}},{{item.age}}
<ul id="app"> 循环输出数字数组 <li v-for="item in sortItems">{{item}}</li> <hr /> 循环输出josn数组 <li v-for="item,index in sortJsonImtes">{{index+1}}.{{item.name}},{{item.age}}</li> </ul> 
 
            var vue = new Vue({
                el: '#app',
                data: { 
                    items:[5,81,2,57,3,9],
                    itemsJson:[
                        {name:'小龙',age:12},
                        {name:'小杨',age:32},
                        {name:'小火',age:22},
                        {name:'小大',age:17}
                    ]
                }, 
                computed:{
                    sortItems:function(){
                        return this.items.sort(function(a,b){
                            return a-b;
                        });
                    },
                    sortJsonImtes:function(){
                        return this.itemsJson.sort(function(a,b){
                            return (a.age-b.age) > 0 ? 1:(a.age-b.age)<0?-1:0; 
                        })
                    }
                }
            })
